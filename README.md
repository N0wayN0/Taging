    $ update /path/to/file tag1 tag2 tag3 "tag 4"

    $ 'insert' 'filepath' 'tags'
    
    $ 'search' 'tags'
    
    $ 'get_tags' 'file'
                  'insert' '/path/to/file' 'tags'
                  'add' '/path/to/file' 'tags'
                  'search' tag OR tag2 OR 'tag 3' AND tag4
                  'in_path' 'cos'
                  'stat_dir' full_path
                  'get_tags' 'file'
                  'update_path' '/path/to/file'
                  'remove' '/path/to/file'
                  'show_all' 'cos'
